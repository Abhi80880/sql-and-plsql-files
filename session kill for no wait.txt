SELECT 'alter system kill session '''|| s.sid|| ','|| s.serial#|| ''' ;' kill_sid
FROM gv$session s
JOIN gv$process p ON p.addr = s.paddr AND p.inst_id = s.inst_id
WHERE s.type != 'BACKGROUND';

SELECT 'alter system kill session '''|| s.sid|| ','|| s.serial#|| ''' ;' kill_sid
FROM 
    v$session s
JOIN 
    v$lock l ON s.sid = l.sid
JOIN 
    dba_objects o ON l.id1 = o.object_id
WHERE 
    l.type = 'TX' -- Transaction locks
    AND l.lmode = 6 -- Row-X lock
ORDER BY 
    s.sid;
